*Write A C Program to swap 2 values*

    int a,b,temp;
    printf("Enter the Number:");
    scanf("%d%d",&a,&b);
    temp=a;
    a=b;
    b=temp;
    printf("Swapped Values Are:");
    printf("a=%d,b=%d",a,b);
}

*Write a C program to find the greatest of 3 numbers*

#include<stdio.h>
#include<conio.h>
int main() {
    int a,b,c;
    printf("Enter Three Numbers:");
    scanf("%d%d%d",&a,&b,&c);
    if(a>b && a>c) {
        printf("%d is Largest",a);
    }
    if(b>a && b>c) {
        printf("%d is Largest",b);
    }
    if(c>a && c>b) {
        printf("%d is Largest",c);
    }
    if(a==b && a==c) {
        printf("All Are Equal");
    }
    return 0;
}
